    <button id="connectBtn" class="sys-btn">🔗 連接智慧手環</button>

    <script>
        const SERVICE_UUID = "4fafc201-1fb5-459e-8fcc-c5c9c331914b";
        const CHARACTERISTIC_UUID = "beb5483e-36e1-4688-b7f5-ea07361b26a8";

        let bluetoothDevice;
        let characteristic;

        document.getElementById('connectBtn').addEventListener('click', async () => {
            try {
                document.getElementById('status').innerText = "正在搜尋藍牙裝置...";
                bluetoothDevice = await navigator.bluetooth.requestDevice({
                    filters: [{ name: 'LifeGuard_Pro' }],
                    optionalServices: [SERVICE_UUID]
                });
                const server = await bluetoothDevice.gatt.connect();
                const service = await server.getPrimaryService(SERVICE_UUID);
                
                // 取得特徵值 (現在同時具備 Notify 與 Write 權限)
                characteristic = await service.getCharacteristic(CHARACTERISTIC_UUID);
                await characteristic.startNotifications();
                characteristic.addEventListener('characteristicvaluechanged', handleData);
                
                document.getElementById('status').innerText = "✅ 連線成功！";
                document.getElementById('connectBtn').innerText = "已連線";
                document.getElementById('connectBtn').style.background = "#28a745";

                bluetoothDevice.addEventListener('gattserverdisconnected', () => {
                    document.getElementById('status').innerText = "💔 手環已斷線，請重新連接！";
                    document.getElementById('connectBtn').innerText = "🔗 重新連接";
                    document.getElementById('connectBtn').style.background = "linear-gradient(90deg, #ff416c 0%, #ff4b2b 100%)";
                });
            } catch (error) {
                document.getElementById('status').innerText = "❌ 連線失敗：" + error.message;
            }
        });

        // 處理接收到的資料 (解析雙模式的不同標籤)
        function handleData(event) {
            const dataString = new TextDecoder('utf-8').decode(event.target.value); 
            const parts = dataString.split(',');
            
            parts.forEach(part => {
                // 模式 1 的標籤解析
                if(part.startsWith('R:')) document.getElementById('val1').innerText = part.substring(2);
                if(part.startsWith('P:')) document.getElementById('val2').innerText = part.substring(2);
                if(part.startsWith('H:')) document.getElementById('heartRate').innerText = part.substring(2);
                
                // 模式 2 的標籤解析
                if(part.startsWith('S:')) document.getElementById('val1').innerText = part.substring(2);
                if(part.startsWith('W:')) document.getElementById('val2').innerText = part.substring(2);
            });
        }

        // 發送切換模式指令給 ESP32，並更新 UI
        async function setMode(mode) {
            // UI 按鈕樣式切換
            document.getElementById('btnMode1').classList.remove('active');
            document.getElementById('btnMode2').classList.remove('active');
            document.getElementById('btnMode' + mode).classList.add('active');

            // 更改卡片標題與顯示狀態
            if (mode === 1) {
                document.getElementById('title1').innerText = "翻滾角 (Roll)";
                document.getElementById('unit1').innerText = "度 (Deg)";
                document.getElementById('title2').innerText = "俯仰角 (Pitch)";
                document.getElementById('unit2').innerText = "度 (Deg)";
                document.getElementById('card-heart').style.display = "block";
            } else {
                document.getElementById('title1').innerText = "瞬時速率 (SPD)";
                document.getElementById('unit1').innerText = "公尺/秒 (m/s)";
                document.getElementById('title2').innerText = "爆發力 (PWR)";
                document.getElementById('unit2').innerText = "徑度/秒 (rad/s)";
                document.getElementById('card-heart').style.display = "none"; // 隱藏心率卡片
                // 切換時先將數值歸零
                document.getElementById('val1').innerText = "0.0";
                document.getElementById('val2').innerText = "0.0";
            }

            // 發送藍牙寫入指令
            if (characteristic) {
                try {
                    const encoder = new TextEncoder();
                    await characteristic.writeValue(encoder.encode(mode.toString()));
                } catch (err) {
                    console.error("切換模式失敗", err);
                }
            }
        }
    </script>
